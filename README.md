```javascript
const config = {

    },
    extensions: [
      '.js', '.jsx', '.ts', '.tsx'
    ],
    modules: [
      path.resolve(__dirname, 'my_modules'),
      path.resolve(__dirname, 'node_modules')
    ],
  },
};
```
